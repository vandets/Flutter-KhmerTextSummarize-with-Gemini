  bool isKhmerOrEnglish(String text) {
    final khmerRegex = RegExp(r'[\u1780-\u17FF]');
    final englishRegex = RegExp(r'[A-Za-z]');
    return khmerRegex.hasMatch(text) || englishRegex.hasMatch(text);
  }

  /// Retry wrapper for API POST requests
  Future<http.Response?> _retryPost(Uri url, Map<String, dynamic> body,
      {int retries = 3, int delayMs = 1000}) async {
    for (int i = 0; i < retries; i++) {
      try {
        final response = await http
            .post(url, headers: {'Content-Type': 'application/json; charset=UTF-8'}, body: jsonEncode(body))
            .timeout(const Duration(seconds: 60));
        return response;
      } catch (e) {
        if (i == retries - 1) rethrow;
        await Future.delayed(Duration(milliseconds: delayMs));
      }
    }
    return null;
  }

  Future<void> summarizeText(String text) async {
    if (text.trim().isEmpty) {
      showMessage('·ûü·ûº·ûò·ûî·ûâ·üí·ûÖ·ûº·ûõ·û¢·ûè·üí·ûê·ûî·ûë...');
      return;
    }

    setState(() {
      _loading = true;
      summaryText = "";
    });

    try {
      if (selectedSummarizer == 'Gemini') {
        final url = Uri.parse('$baseUrl/api/summarize');
        final response = await _retryPost(url, {'text': text});
        if (response != null && response.statusCode == 200) {
          final decoded = utf8.decode(response.bodyBytes);
          final data = jsonDecode(decoded);
          setState(() => summaryText = data['summary'] ?? "");
        } else {
          showMessage('Error: ${response?.statusCode ?? "No response"}');
        }
      } else if (selectedSummarizer == 'Local Khmer') {
        final summary = await summarizeTextLocally(text);
        setState(() => summaryText = summary);
      }
    } catch (e) {
      showMessage('Exception: $e');
    } finally {
      setState(() => _loading = false);
    }
  }

  Future<String> summarizeTextLocally(String text) async {
    // Stub for your local summarization logic or backend
    final url = Uri.parse('$baseUrl/api/local-summarize/text');
    final response = await _retryPost(url, {'text': text});
    if (response != null && response.statusCode == 200) {
      final data = jsonDecode(response.body);
      return data['summary'] ?? '';
    } else {
      return 'Failed to summarize locally';
    }
  }


  Future<void> uploadPdf(File file) async {
    setState(() => _loading = true);
    showMessage("·ûÄ·üÜ·ûñ·ûª·ûÑ·ûä·üÜ·ûé·ûæ·ûö·ûÄ·û∂·ûö PDF...");

    try {
      // Extract text
      final bytes = await file.readAsBytes();
      final document = PdfDocument(inputBytes: bytes);
      String text = PdfTextExtractor(document).extractText();
      document.dispose();

      if (text.trim().isEmpty) {
        // Fallback to direct upload
        await uploadPdfDirect(file);
        return;
      }

      // Optional truncation
      if (text.length > 50000) text = text.substring(0, 50000) + " [TEXT TRUNCATED]";

      // Send extracted text to summarize API
      await summarizeText(text);
    } catch (e) {
      showMessage('Exception: $e');
    } finally {
      setState(() => _loading = false);
    }
  }

  Future<void> uploadPdfDirect(File file) async {
    summaryText = "";
    setState(() => _loading = true);

    try {
      final uri = Uri.parse('$baseUrl/api/summarize/upload-direct');
      final bytes = await file.readAsBytes();
      final base64Pdf = base64Encode(bytes);

      final payload = jsonEncode({
        "inputDocuments": [
          {"mimeType": "application/pdf", "data": base64Pdf},
        ],
        "contents": [
          {
            "parts": [
              {"text": "·ûó·û∂·ûö·ûÄ·û∑·ûÖ·üí·ûÖ: ·ûü·ûº·ûò·ûü·ûÑ·üí·ûÅ·üÅ·ûî·û¢·ûè·üí·ûê·ûî·ûë·ûÄ·üí·ûì·ûª·ûÑ PDF ·ûì·üÅ·üá·ûá·û∂·ûó·û∂·ûü·û∂·ûÅ·üí·ûò·üÇ·ûö·üî"},
            ],
          },
        ],
      });

      final response = await _retryPost(uri, payload as Map<String, dynamic>);
      if (response != null && response.statusCode == 200) {
        final data = jsonDecode(response.body);
        setState(() => summaryText = data['summary'] ?? "");
      } else {
        showMessage('Error: ${response?.statusCode ?? "No response"}');
      }
    } catch (e) {
      showMessage('Exception: $e');
    } finally {
      setState(() => _loading = false);
    }
  }

  Future<void> pickFileAndSummarize() async {
    summaryText = "";
    FilePickerResult? result = await FilePicker.platform.pickFiles(type: FileType.custom, allowedExtensions: ['txt', 'pdf']);
    if (result != null && result.files.single.path != null) {
      File file = File(result.files.single.path!);
      try {
        if (file.path.endsWith('.txt')) {
          String content = await file.readAsString();
          if (content.trim().isEmpty) {
            showMessage('The file is empty or cannot be read.');
            return;
          }
          if (!isKhmerOrEnglish(content)) {
            showMessage('Only Khmer or English text allowed.');
            return;
          }
          _controller.text = content;
          await summarizeText(content);
        } else if (file.path.endsWith('.pdf')) {
          await uploadPdf(file);
        } else {
          showMessage('File format not supported.');
        }
      } catch (e) {
        showMessage('Failed to read file: $e');
      }
    } else {
      showMessage('No file selected.');
    }
  }

  Future<void> _downloadText(BuildContext context) async {
    if (summaryText.isEmpty) {
      showMessage("No summary to save.");
      return;
    }
    final directory = await getApplicationDocumentsDirectory();
    final file = File('${directory.path}/summary.txt');
    await file.writeAsString(summaryText);
    showMessage("Saved as summary.txt in ${directory.path}");
  }

// Count Khmer words
  int countKhmerWords(String text) {
    final wordRegex = RegExp(r'[\u1780-\u17FF]+');
    return wordRegex.allMatches(text).length;
  }

// Count sentences
  int countSentences(String text) {
    final sentenceEndings = RegExp(r'[·üî.!?]+');
    final sentences = text.split(sentenceEndings);
    return sentences.where((s) => s.trim().isNotEmpty).length;
  }

  int countPunctuation(String text) {
    // Use normal string for non-ASCII characters
    final punctuation = RegExp('[.,!?;:()\"‚Äú‚Äù\'\\[\\]{}\\-·üî·üï]');
    return punctuation.allMatches(text).length;
  }

// 2. New Method to Handle Language Detection (Debounced)
  // You'll need to update the existing _handleLanguageDetection function

  void _handleLanguageDetection(String text) {
    if (_debounce?.isActive ?? false) _debounce!.cancel();

    _debounce = Timer(const Duration(milliseconds: 500), () async {
      if (text.isEmpty) {
        setState(() {
          detectedLanguage = '';
          wordCount = 0; // Reset word count
        });
        return;
      }

      try {
        // 1. Language Detection
        final langResult = await detectLanguageWithGemini(text);

        // 2. Word Count (using the new function)
        final countResult = await countKhmerWordsWithGemini(text);

        // Update the UI state with both results
        if (mounted) {
          setState(() {
            detectedLanguage = langResult;
            wordCount = countResult; // Update word count
          });
        }
      } catch (e) {
        // Handle API errors gracefully
        if (mounted) {
          setState(() {
            detectedLanguage = 'Error';
            wordCount = 0;
          });
        }
        print('API call failed: $e');
      }
    });
  }

  Future<bool> _isServerReachable() async {
    if (baseUrl.isEmpty) return false;

    try {
      final uri = Uri.parse('$baseUrl/api/summarize');
      final response = await http.get(uri).timeout(const Duration(seconds: 60));
      return response.statusCode == 200;
    } catch (_) {
      return false;
    }
  }
